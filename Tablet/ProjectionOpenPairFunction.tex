\begin{definition}
For projected sizes \(\ell_R,\ell_B\), define
\[
\begin{aligned}
f(\ell_R,\ell_B)
&=\binom{\ell_R}{2}+\binom{\ell_B}{2}\\
&\quad-\min\left\{\binom{k-\ell_R}{2},
\binom{pn}{2}+\binom{k-\ell_R-pn}{2}\right\}\\
&\quad-\min\left\{\binom{k-\ell_B}{2},
\binom{pn}{2}+\binom{k-\ell_B-pn}{2}\right\}.
\end{aligned}
\]
This is the lower-bound expression for the number of open projected pairs.
\end{definition}
