\begin{helper}
For all \(r,t\in\mathbb R\),
\[
\noderef{RealChooseTwo}(r+t)
-\noderef{RealChooseTwo}(r)
-\noderef{RealChooseTwo}(t)
=r\,t.
\]
\end{helper}
\begin{proof}
Write \(C(u)=\noderef{RealChooseTwo}(u)=u(u-1)/2\).  Then
\[
C(r+t)-C(r)-C(t)
=\frac{(r+t)(r+t-1)-r(r-1)-t(t-1)}{2}.
\]
Expanding the numerator and collecting terms gives \(2rt\), so the displayed
quantity is \(rt\).
\end{proof}
